```python
@router.post("/portal", response_model=dict)
def get_portal_url(
    db: Session = Depends(get_db),
    current_user: Usuario = Depends(get_current_user),
):
    """
    Retorna a URL do Stripe Customer Portal para o usuário gerenciar
    pagamento/assinatura. Requer gateway Stripe com chave privada configurada.
    """
    from app.models.pagamento import Pagamento

    gw = configuracao_service.obter_gateway_raw(db)
    if gw.get("gateway_tipo") != "stripe" or not gw.get("gateway_chave_privada"):
        raise HTTPException(status_code=400, detail="Gateway Stripe não configurado.")

    # Find most recent active stripe subscription to get customer id
    pag = (
        db.query(Pagamento)
        .filter(
            Pagamento.usuario_id == current_user.id,
            Pagamento.status == "aprovado",
            Pagamento.referencia_externa.isnot(None),
        )
        .order_by(Pagamento.criado_em.desc())
        .first()
    )
    if not pag:
        raise HTTPException(status_code=404, detail="Nenhuma assinatura ativa encontrada.")

    try:
        import stripe
        stripe.api_key = gw.get("gateway_chave_privada")
        # Retrieve subscription to get customer id
        sub = stripe.Subscription.retrieve(pag.referencia_externa)
        customer_id = sub["customer"]
        session = stripe.billing_portal.Session.create(
            customer=customer_id,
            return_url=f"{settings.FRONTEND_URL}/dashboard/cobranca/pagamento",
        )
        return {"url": session.url}
    except Exception as e:
        raise HTTPException(status_code=502, detail=f"Erro ao criar portal Stripe: {e}")
```

**Note:** Requires `settings.FRONTEND_URL` — check `config.py` for the existing env var name (use `FRONTEND_URL` or the existing equivalent). Also requires the `Pagamento` model import path is correct (already in `cobranca.py`'s webhook handler so re-use that pattern).

```python
from sqlalchemy import Column, Integer, String, Boolean, Float, DateTime, Enum
from app.db.database import Base
import enum


class TipoDesconto(str, enum.Enum):
    percentual = "percentual"
    fixo = "fixo"


class Cupom(Base):
    __tablename__ = "cupons"

    id = Column(Integer, primary_key=True, index=True)
    codigo = Column(String(50), unique=True, nullable=False, index=True)
```

```python
from datetime import datetime, timezone
from fastapi import APIRouter, Depends, HTTPException
from pydantic import BaseModel
from sqlalchemy.orm import Session
from typing import Optional

from app.db.database import get_db
from app.models.cupom import Cupom
from app.models.usuario import Usuario
from app.core.deps import get_current_user

router = APIRouter(tags=["cupons"])


# ── Schemas (inline to avoid extra files) ──────────────────────────────────────

class CupomCreate(BaseModel):
    codigo: str
    tipo_desconto: str          # "percentual" | "fixo"
    valor_desconto: float
    valido_ate: Optional[datetime] = None
    usos_max: Optional[int] = None

class CupomUpdate(BaseModel):
    ativo: Optional[bool] = None
    valido_ate: Optional[datetime] = None
    usos_max: Optional[int] = None

class CupomOut(BaseModel):
    id: int
    codigo: str
    tipo_desconto: str
    valor_desconto: float
    valido_ate: Optional[datetime]
    usos_max: Optional[int]
    usos_atual: int
    ativo: bool
    criado_em: datetime

    class Config:
        from_attributes = True

class ValidarCupomRequest(BaseModel):
    codigo: str

class ValidarCupomResponse(BaseModel):
    valido: bool
    tipo_desconto: Optional[str] = None
    valor_desconto: Optional[float] = None
    mensagem: str


# ── Admin endpoints ────────────────────────────────────────────────────────────

def _require_admin(user: Usuario) -> None:
    if user.perfil != "admin":
        raise HTTPException(status_code=403, detail="Acesso restrito a administradores.")


@router.post("/admin/cupons/", response_model=CupomOut, status_code=201)
def criar_cupom(
    body: CupomCreate,
    db: Session = Depends(get_db),
    current_user: Usuario = Depends(get_current_user),
):
    _require_admin(current_user)
    if db.query(Cupom).filter_by(codigo=body.codigo.upper()).first():
        raise HTTPException(status_code=409, detail="Código já existe.")
    cupom = Cupom(
        codigo=body.codigo.upper(),
        tipo_desconto=body.tipo_desconto,
        valor_desconto=body.valor_desconto,
        valido_ate=body.valido_ate,
        usos_max=body.usos_max,
    )
    db.add(cupom)
    db.commit()
    db.refresh(cupom)
    return cupom


@router.get("/admin/cupons/", response_model=list[CupomOut])
def listar_cupons(
    db: Session = Depends(get_db),
    current_user: Usuario = Depends(get_current_user),
):
    _require_admin(current_user)
    return db.query(Cupom).order_by(Cupom.criado_em.desc()).all()


@router.patch("/admin/cupons/{cupom_id}", response_model=CupomOut)
def atualizar_cupom(
    cupom_id: int,
    body: CupomUpdate,
    db: Session = Depends(get_db),
    current_user: Usuario = Depends(get_current_user),
):
    _require_admin(current_user)
    cupom = db.query(Cupom).filter_by(id=cupom_id).first()
    if not cupom:
        raise HTTPException(status_code=404, detail="Cupom não encontrado.")
    if body.ativo is not None:
        cupom.ativo = body.ativo
    if body.valido_ate is not None:
        cupom.valido_ate = body.valido_ate
    if body.usos_max is not None:
        cupom.usos_max = body.usos_max
    db.commit()
    db.refresh(cupom)
    return cupom


# ── Public (authenticated) endpoint ───────────────────────────────────────────

@router.post("/cupons/validar", response_model=ValidarCupomResponse)
def validar_cupom(
    body: ValidarCupomRequest,
    db: Session = Depends(get_db),
    _: Usuario = Depends(get_current_user),
):
    cupom = db.query(Cupom).filter_by(codigo=body.codigo.upper(), ativo=True).first()
    if not cupom:
        return ValidarCupomResponse(valido=False, mensagem="Cupom inválido ou inativo.")
    now = datetime.now(timezone.utc)
    if cupom.valido_ate and cupom.valido_ate < now:
        return ValidarCupomResponse(valido=False, mensagem="Cupom expirado.")
    if cupom.usos_max is not None and cupom.usos_atual >= cupom.usos_max:
        return ValidarCupomResponse(valido=False, mensagem="Cupom esgotado.")
    return ValidarCupomResponse(
        valido=True,
        tipo_desconto=cupom.tipo_desconto,
        valor_desconto=cupom.valor_desconto,
        mensagem="Cupom válido.",
    )
```

```tsx
interface QuotaBarProps {
  used: number;
  max: number | null;
  /** Additional className for the outer wrapper */
  className?: string;
}

/**
 * Simple horizontal progress bar for quota display.
 * Turns amber at 70%, red at 90%. Pulses green when unlimited (max=null).
 */
export default function QuotaBar({ used, max, className = "" }: QuotaBarProps) {
  const pct = max === null ? 0 : Math.min(100, (used / max) * 100);
  const danger = max !== null && pct >= 90;
  const warn = max !== null && pct >= 70;
  const barColor = danger ? "bg-red-500" : warn ? "bg-amber-400" : "bg-blue-500";

  return (
    <div className={`w-full h-2 bg-slate-100 rounded-full overflow-hidden ${className}`}>
      {max === null ? (
        <div className="h-full bg-emerald-400 w-full opacity-30 animate-pulse" />
      ) : (
        <div
          className={`h-full rounded-full transition-all duration-700 ${barColor}`}
          style={{ width: `${pct}%` }}
        />
      )}
    </div>
  );
}
```

```tsx
"use client";
import { useState, useEffect } from "react";
import Link from "next/link";
import { useAuth } from "../../../context/AuthContext";

const API = process.env.NEXT_PUBLIC_API_URL ?? "http://localhost:40001";

export default function PagamentoPage() {
  const { user } = useAuth();
  const [assinatura, setAssinatura] = useState<any>(null);
  const [loading, setLoading] = useState(true);
  const [redirecting, setRedirecting] = useState(false);
  const [error, setError] = useState("");

  useEffect(() => {
    const token = localStorage.getItem("bai_token");
    if (!token) return;
    fetch(`${API}/api/v1/cobranca/assinatura`, {
      headers: { Authorization: `Bearer ${token}` },
    })
      .then((r) => r.ok ? r.json() : null)
      .then((d) => { setAssinatura(d); setLoading(false); })
      .catch(() => setLoading(false));
  }, []);

  async function handlePortal() {
    setRedirecting(true);
    setError("");
    const token = localStorage.getItem("bai_token");
    try {
      const res = await fetch(`${API}/api/v1/cobranca/portal`, {
        method: "POST",
        headers: { Authorization: `Bearer ${token}` },
      });
      if (!res.ok) {
        const d = await res.json();
        setError(d.detail ?? "Erro ao abrir portal.");
        setRedirecting(false);
        return;
      }
      const { url } = await res.json();
      window.location.href = url;
    } catch {
      setError("Erro de rede. Tente novamente.");
      setRedirecting(false);
    }
  }

  if (loading) {
    return (
      <div className="flex items-center justify-center py-20">
        <div className="animate-spin w-8 h-8 border-4 border-blue-600 border-t-transparent rounded-full" />
      </div>
    );
  }

  return (
    <div className="max-w-lg mx-auto py-10 px-4">
      <div className="mb-6">
        <Link href="/dashboard/cobranca" className="text-sm text-slate-500 hover:text-slate-700">
          ← Voltar para Plano & Consumo
        </Link>
      </div>

      <h1 className="text-2xl font-black text-slate-900 mb-1">Método de Pagamento</h1>
      <p className="text-slate-500 text-sm mb-8">
        Gerencie sua assinatura e cartão de crédito via portal seguro do Stripe.
      </p>

      <div className="bg-white rounded-3xl border border-slate-100 shadow-sm p-6 mb-6">
        <div className="text-xs font-bold text-slate-400 uppercase tracking-widest mb-3">Plano atual</div>
        <div className="text-lg font-black text-slate-900 capitalize">
          {user?.tipo_plano ?? "Gratuito"}
        </div>
        {assinatura?.status && (
          <div className="text-sm text-slate-500 mt-1">
            Status: <span className="font-semibold text-slate-700">{assinatura.status}</span>
          </div>
        )}
        {assinatura?.expira_em && (
          <div className="text-sm text-slate-500 mt-1">
            Próxima renovação:{" "}
            <span className="font-semibold text-slate-700">
              {new Date(assinatura.expira_em).toLocaleDateString("pt-BR")}
            </span>
          </div>
        )}
      </div>

      {error && (
        <div className="bg-red-50 text-red-700 text-sm rounded-2xl px-4 py-3 mb-4">{error}</div>
      )}

      {user?.tipo_plano && user.tipo_plano !== "gratuito" ? (
        <button
          onClick={handlePortal}
          disabled={redirecting}
          className="w-full bg-blue-600 hover:bg-blue-700 disabled:opacity-60 text-white font-bold py-3 rounded-xl text-sm transition-colors"
        >
          {redirecting ? "Redirecionando..." : "Trocar método de pagamento"}
        </button>
      ) : (
        <Link
          href="/dashboard/cobranca/upgrade"
          className="w-full block text-center bg-blue-600 hover:bg-blue-700 text-white font-bold py-3 rounded-xl text-sm transition-colors"
        >
          Assinar um plano pago
        </Link>
      )}

      <p className="text-xs text-slate-400 text-center mt-4">
        Você será redirecionado para o portal seguro do Stripe. Nenhum dado de cartão passa pelo nosso servidor.
      </p>
    </div>
  );
}
```

### Step 15 — Create WhatsApp dashboard page

**File (new):** `D:/Sources/BAI/frontend/src/app/dashboard/whatsapp/page.tsx`

**Content:**

```tsx
"use client";
import { useState, useEffect, useRef } from "react";
import { useAuth } from "../../context/AuthContext";
import PlanGate from "@/app/components/PlanGate";

const API = process.env.NEXT_PUBLIC_API_URL ?? "http://localhost:40001";
const POLL_INTERVAL = 3000; // ms

type SessionStatus = { status: string; qr: string | null };
type Message = { id: number; jid_conversa: string; direcao: string; conteudo: string; enviado_em: string };

export default function WhatsAppPage() {
  const { user } = useAuth();
  const [session, setSession] = useState<SessionStatus | null>(null);
  const [loadingSession, setLoadingSession] = useState(true);
  const [messages, setMessages] = useState<Message[]>([]);
  const [sendTo, setSendTo] = useState("");
  const [sendText, setSendText] = useState("");
  const [sending, setSending] = useState(false);
  const [sendError, setSendError] = useState("");
  const pollRef = useRef<ReturnType<typeof setInterval> | null>(null);

  // Feature gate: brokers/agencies without whatsapp plan
  const isBlocked =
    user &&
    ["corretor", "imobiliaria"].includes(user.role === "broker" ? "corretor" : user.role === "agency" ? "imobiliaria" : user.role) &&
    !user.features?.whatsapp;

  // Note: `user.features` must be populated from AuthContext — see Step 16.
  // If AuthContext doesn't expose features yet, read from consumo endpoint instead:
  // const [blocked, setBlocked] = useState(false); // fetch /cobranca/consumo on mount

  function token() { return localStorage.getItem("bai_token") ?? ""; }

  async function fetchStatus() {
    const res = await fetch(`${API}/api/v1/whatsapp/session/status`, {
      headers: { Authorization: `Bearer ${token()}` },
    });
    if (res.ok) setSession(await res.json());
  }

  async function startSession() {
    await fetch(`${API}/api/v1/whatsapp/session/start`, {
      method: "POST",
      headers: { Authorization: `Bearer ${token()}` },
    });
    fetchStatus();
  }

  async function stopSession() {
    await fetch(`${API}/api/v1/whatsapp/session`, {
      method: "DELETE",
      headers: { Authorization: `Bearer ${token()}` },
    });
    setSession(null);
  }

  async function fetchMessages() {
    const res = await fetch(`${API}/api/v1/whatsapp/messages?limit=50`, {
      headers: { Authorization: `Bearer ${token()}` },
    });
    if (res.ok) setMessages(await res.json());
  }

  async function sendMessage(e: React.FormEvent) {
    e.preventDefault();
    if (!sendTo.trim() || !sendText.trim()) return;
    setSending(true);
    setSendError("");
    const res = await fetch(`${API}/api/v1/whatsapp/send`, {
      method: "POST",
      headers: { Authorization: `Bearer ${token()}`, "Content-Type": "application/json" },
      body: JSON.stringify({ to: sendTo.trim(), text: sendText.trim() }),
    });
    if (res.ok) {
      setSendText("");
      fetchMessages();
    } else {
      const d = await res.json();
      setSendError(d.detail ?? "Erro ao enviar.");
    }
    setSending(false);
  }

  useEffect(() => {
    fetchStatus().finally(() => setLoadingSession(false));
    fetchMessages();
    pollRef.current = setInterval(() => {
      fetchStatus();
      fetchMessages();
    }, POLL_INTERVAL);
    return () => { if (pollRef.current) clearInterval(pollRef.current); };
  }, []);

  // If plan doesn't include whatsapp, show gate
  // Simple approach: if 402 comes back from session/status, show gate.
  // Controlled via `isBlocked` derived from user context.

  return (
    <PlanGate featureName="WhatsApp" minPlan="Pro" blocked={!!isBlocked}>
      <div className="max-w-2xl mx-auto py-8 px-4 space-y-6">
        <h1 className="text-2xl font-black text-slate-900">WhatsApp</h1>

        {/* Session panel */}
        <div className="bg-white rounded-3xl border border-slate-100 shadow-sm p-6">
          <div className="flex items-center justify-between mb-4">
            <div>
              <div className="text-xs font-bold text-slate-400 uppercase tracking-widest mb-1">Status da Sessão</div>
              <div className="text-sm font-bold text-slate-700 capitalize">
                {loadingSession ? "Verificando..." : session?.status ?? "Desconectado"}
              </div>
            </div>
            {session?.status === "connected" ? (
              <button
                onClick={stopSession}
                className="text-sm font-bold text-red-600 hover:text-red-800 px-4 py-2 border border-red-200 rounded-xl transition-colors"
              >
                Desconectar
              </button>
            ) : (
              <button
                onClick={startSession}
                className="text-sm font-bold text-white bg-green-600 hover:bg-green-700 px-4 py-2 rounded-xl transition-colors"
              >
                Conectar
              </button>
            )}
          </div>

          {/* QR code */}
          {session?.qr && session.status !== "connected" && (
            <div className="mt-4 flex flex-col items-center gap-3">
              <p className="text-sm text-slate-500">Escaneie o QR code com o WhatsApp no celular</p>
              {/* eslint-disable-next-line @next/next/no-img-element */}
              <img src={session.qr} alt="QR Code WhatsApp" className="w-48 h-48 rounded-2xl border border-slate-100" />
            </div>
          )}
        </div>

        {/* Send message */}
        {session?.status === "connected" && (
          <div className="bg-white rounded-3xl border border-slate-100 shadow-sm p-6">
            <div className="text-xs font-bold text-slate-400 uppercase tracking-widest mb-4">Enviar Mensagem</div>
            <form onSubmit={sendMessage} className="space-y-3">
              <input
                type="text"
                value={sendTo}
                onChange={(e) => setSendTo(e.target.value)}
                placeholder="Número (ex: 5511999999999)"
                className="w-full border border-slate-200 rounded-xl px-4 py-2.5 text-sm focus:outline-none focus:ring-2 focus:ring-blue-500"
              />
              <textarea
                value={sendText}
                onChange={(e) => setSendText(e.target.value)}
                rows={3}
                placeholder="Mensagem"
                className="w-full border border-slate-200 rounded-xl px-4 py-2.5 text-sm focus:outline-none focus:ring-2 focus:ring-blue-500 resize-none"
              />
              {sendError && <p className="text-sm text-red-600">{sendError}</p>}
              <button
                type="submit"
                disabled={sending}
                className="w-full bg-blue-600 hover:bg-blue-700 disabled:opacity-60 text-white font-bold py-2.5 rounded-xl text-sm transition-colors"
              >
                {sending ? "Enviando..." : "Enviar"}
              </button>
            </form>
          </div>
        )}

        {/* Message list */}
        {messages.length > 0 && (
          <div className="bg-white rounded-3xl border border-slate-100 shadow-sm p-6">
            <div className="text-xs font-bold text-slate-400 uppercase tracking-widest mb-4">Mensagens Recentes</div>
            <div className="space-y-3 max-h-72 overflow-y-auto">
              {messages.map((m) => (
                <div key={m.id} className={`flex ${m.direcao === "saida" ? "justify-end" : "justify-start"}`}>
                  <div
                    className={`max-w-[75%] px-4 py-2 rounded-2xl text-sm ${
                      m.direcao === "saida"
                        ? "bg-blue-600 text-white rounded-br-none"
                        : "bg-slate-100 text-slate-800 rounded-bl-none"
                    }`}
                  >
                    <p>{m.conteudo}</p>
                    <p className={`text-[10px] mt-1 ${m.direcao === "saida" ? "text-blue-200" : "text-slate-400"}`}>
                      {m.jid_conversa.split("@")[0]} · {new Date(m.enviado_em).toLocaleTimeString("pt-BR", { hour: "2-digit", minute: "2-digit" })}
                    </p>
                  </div>
                </div>
              ))}
            </div>
          </div>
        )}
      </div>
    </PlanGate>
  );
}
```

**Note on plan gate:** The cleanest approach is to fetch `GET /api/v1/cobranca/consumo` on mount and check if `whatsapp` feature is in the plan — OR rely on the 402 error from the backend. For now, add a simple `useEffect` fallback: if a 402 is returned from `session/status`, set a local `blocked` state and render `<PlanGate blocked={blocked}>`.

**Dependencies:** Steps 7 (backend gate), 8 (PlanGate component).
